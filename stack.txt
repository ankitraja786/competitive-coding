#include<bits/stdc++.h>
using namespace std;
struct node
{
 int top;
 node* next;
};

node* push(int t)
{
 node* n=new node;
 n->top=t;
 n->next=NULL;
 return n;
}

void pop(node* start)
{
   node* pre=new node;
   node* i=new node;
 pre=start;
 i=pre->next;
 for(;i!=NULL;i=i->next)
   {
     if(i->next==NULL)
     {
      pre->next=NULL;
      break;
     }
    pre=pre->next;
   }

}

 //start=start->next;

 }
}
